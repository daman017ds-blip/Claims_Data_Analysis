import re

def clean_date(x):
    if pd.isna(x):
        return pd.NaT

    x = str(x)
    x = re.sub(r"\s*\(.*\)$", "", x)  # Remove "(Eastern Daylight Time)"
    x = x.replace("GMT", "")          # Remove "GMT"

    return pd.to_datetime(x, errors="coerce")

dataset["Date_of_filing"] = dataset["Date_of_filing"].apply(clean_date)
